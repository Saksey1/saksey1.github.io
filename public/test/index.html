#pragma once	

#include "CoreMinimal.h"
#include "WeaponActor.h"
#include "GameFramework/Character.h"
#include "Enemy.generated.h"

class ASuperpositionCharacter;
UCLASS()
class SUPERPOSITION_API AEnemy : public ACharacter
{
	GENERATED_BODY()
public:
	AEnemy();

	// player reference for chasing
	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="References")
	ASuperpositionCharacter* TargetPlayer;	
	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Player")
	FVector TargetLastKnownLocation;
	UFUNCTION(BlueprintCallable, Category="Player")
	void UpdateLastKnownPlayerLocation(FVector TargetLocation);
	
	//Movement  Variables
	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement|Locations")
	FVector StartLocation;
	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement|Locations")
	FVector EndLocation;
	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement|Locations")
	FVector TargetLocation;

	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement")
	int MoveSpeed = 6.0f;
	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement")
	int MinMoveDistance = 10;
	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement")
	int MaxMoveDistance = 100;

	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement")
	float DistanceThreshold = 50;
	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement")
	int MovementTimeFactor = 2;

	UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Movement")
	bool bIsMoving = true;	

	// Health stuff
	UPROPERTY(BlueprintReadWrite, EditAnywhere, Category="Health")
	class UHealthComponent* HealthComponent;
	
	// Weapon Stuff
	// Range at which enemy has chased enough and can attack
	UPROPERTY(BlueprintReadWrite, EditAnywhere, Category="Attack")
	float MaxAttackRange = 50.f;
	UPROPERTY(BlueprintReadWrite, EditAnywhere, Category="Attack")
	float MinAttackRange = 5.f;
	UPROPERTY(BlueprintReadWrite,EditAnywhere,Category="Attack")
	float MidAttackRange;
	
	UPROPERTY(EditAnywhere,BlueprintReadWrite,Category="Weapon")
	TSubclassOf<AWeaponActor> WeaponClass;
	
	UPROPERTY()
	TArray<AActor*> Weapons;
	
	UFUNCTION()
	void SpawnWeapons();
	
	UFUNCTION()
	void AttemptWeaponFire();

	UFUNCTION()
	void EndWeaponFire();
	
	//In relation to death
	//the time added to the world countdown in *min* after death
	UPROPERTY(BlueprintReadWrite, EditAnywhere, Category="Health")
	float TimeAddedToCountdown = 0.5;
	// MISC Methods
	UFUNCTION(BlueprintCallable, Category="Methods")
	float DistanceFromPoint(FVector Location, AActor* Target);
	
	// Called when the game starts or when spawned
	virtual void BeginPlay() override;

public:
	// Called every frame
	virtual void Tick(float DeltaTime) override;

	virtual void Destroyed() override;
	
	UFUNCTION()
	UHealthComponent* GetHealthComponent() const;
};
